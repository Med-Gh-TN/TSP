# 1. Initialiser
# Commencer avec une solution aléatoire
solution_actuelle = list(range(nombre_villes))
random.shuffle(solution_actuelle)

# La meilleure solution est la solution de départ
meilleure_solution = solution_actuelle[:]

# La liste tabou (mémoire) est vide
tabu_list = deque(maxlen=taille_tabu)
                

# 2. Générer le Voisinage
def generer_voisins(solution):
    """Génère toutes les solutions voisines
    en échangeant chaque paire de villes."""
    voisins = []
    for i in range(len(solution)):
        for j in range(i + 1, len(solution)):
            voisin = solution[:]
            voisin[i], voisin[j] = voisin[j], voisin[i]
            voisins.append(voisin)
    return voisins
                

# 3. Sélectionner le Meilleur Voisin
# Filtrer les voisins qui sont dans la liste tabou
voisins = [v for v in voisins if tuple(v) not in tabu_list]

# Sélectionner le meilleur voisin parmi les non-tabous
meilleur_voisin = min(voisins, key=lambda v: ...)

# Mettre à jour la solution actuelle
solution_actuelle = meilleur_voisin
                

# 4. Mettre à jour la Mémoire et le Meilleur Global
# Ajouter la nouvelle solution à la liste tabou
tabu_list.append(tuple(solution_actuelle))

# Mettre à jour la meilleure solution globale si nécessaire
if distance_actuelle < meilleure_distance:
    meilleure_solution = solution_actuelle[:]
    meilleure_distance = distance_actuelle
